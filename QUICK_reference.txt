│ USEFUL SQL QUERIES                                                    │
└───────────────────────────────────────────────────────────────────────┘

-- View top recommendations
SELECT p.name, s.name, r.score, r.rationale 
FROM recommendation r
JOIN product p ON r.product_id = p.product_id
JOIN store_section s ON r.recommended_section_id = s.section_id
ORDER BY r.score DESC LIMIT 10;

-- Section traffic
SELECT section_id, COUNT(*) as visits 
FROM movement 
GROUP BY section_id 
ORDER BY visits DESC;

-- Community breakdown
SELECT community_id, COUNT(*) as sections, GROUP_CONCAT(section_id) 
FROM section_community 
GROUP BY community_id;

-- Product cluster sizes
SELECT cluster_id, COUNT(*) as products 
FROM product_cluster 
GROUP BY cluster_id;
